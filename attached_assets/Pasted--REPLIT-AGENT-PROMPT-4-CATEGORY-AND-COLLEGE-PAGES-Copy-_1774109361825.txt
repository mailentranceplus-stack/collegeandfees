### PAGE 2: `pages/colleges/[college-slug].js` — COLLEGE OVERVIEW PAGE

This is the main college page. Tab bar shows **Overview** as active.

**getServerSideProps:**
```js
export async function getServerSideProps({ params }) {
  const slug = params['college-slug']

  const { data: college } = await supabase
    .from('colleges')
    .select('*')
    .eq('slug', slug)
    .single()

  if (!college) return { notFound: true }

  // College content (SEO text, highlights, about)
  const { data: content } = await supabase
    .from('college_content')
    .select('*')
    .eq('college_id', college.id)
    .single()

  // Latest placements
  const { data: placements } = await supabase
    .from('placements')
    .select('*')
    .eq('college_id', college.id)
    .order('year', { ascending: false })
    .limit(1)
    .single()

  // NIRF ranking
  const { data: ranking } = await supabase
    .from('rankings')
    .select('rank, year')
    .eq('college_id', college.id)
    .eq('source', 'NIRF')
    .order('year', { ascending: false })
    .limit(1)
    .single()

  // Summary fee info (management quota only, latest year, first few courses)
  const { data: fees } = await supabase
    .from('fees')
    .select(`tuition_fee, quota, courses(name, short_name)`)
    .eq('college_id', college.id)
    .eq('quota', 'management')   // DB value is lowercase 'management'
    .eq('academic_year', '2024-25')
    .order('tuition_fee', { ascending: false })
    .limit(5)

  // Courses offered
  const { data: courses } = await supabase
    .from('college_courses')
    .select(`
      total_seats, mgmt_quota_seats, govt_quota_seats,
      courses(name, short_name, degree)
    `)
    .eq('college_id', college.id)

  // Facilities
  const { data: facilities } = await supabase
    .from('college_facilities')
    .select(`facilities(name, icon)`)
    .eq('college_id', college.id)

  // FAQs
  const { data: faqs } = await supabase
    .from('faqs')
    .select('question, answer')
    .eq('college_id', college.id)
    .eq('is_active', true)
    .order('sort_order')
    .limit(6)

  // Similar colleges
  const { data: similarColleges } = await supabase
    .from('colleges')
    .select('id, slug, name, short_name, naac_grade')
    .eq('city', college.city)
    .eq('is_active', true)
    .neq('id', college.id)
    .limit(4)

  return {
    props: {
      college,
      content: content || null,
      placements: placements || null,
      ranking: ranking || null,
      fees: fees || [],
      courses: courses || [],
      facilities: facilities || [],
      faqs: faqs || [],
      similarColleges: similarColleges || [],
    }
  }
}
```

### PAGE 3: `pages/colleges/[college-slug]/fees.js` — FEES SUB-PAGE

This requires a nested dynamic route: `pages/colleges/[college-slug]/fees.js`

**The slug is in the URL as `[college-slug]` — access via `params['college-slug']`**

**getServerSideProps:**
```js
export async function getServerSideProps({ params }) {
  const slug = params['college-slug']

  const { data: college } = await supabase
    .from('colleges')
    .select('*')
    .eq('slug', slug)
    .single()

  if (!college) return { notFound: true }

  // ALL fees for this college, latest academic year, ALL quotas
  const { data: fees } = await supabase
    .from('fees')
    .select(`
      tuition_fee, hostel_fee, other_fees, total_fee, academic_year, quota,
      courses(id, name, short_name, degree)
    `)
    .eq('college_id', college.id)
    .eq('academic_year', '2024-25')
    .order('quota', { ascending: true })
    .order('tuition_fee', { ascending: false })

  const { data: content } = await supabase
    .from('college_content')
    .select('meta_title, meta_desc')
    .eq('college_id', college.id)
    .single()

  const { data: ranking } = await supabase
    .from('rankings')
    .select('rank')
    .eq('college_id', college.id)
    .eq('source', 'NIRF')
    .order('year', { ascending: false })
    .limit(1)
    .single()

  return {
    props: { college, fees: fees || [], content: content || null, ranking: ranking || null }
  }
}
```
